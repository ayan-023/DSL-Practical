def insert(key,value,hash_table,size):
    index=key%size
    while hash_table[index] is not None:
        index=(index+1)%size
    hash_table[index]=(key,value)
    
    
def delete(key,hash_table,size):
    index=key%size
    for _ in range(size):
        
        if key in hash_table[index]:
            print("deleted :: ",hash_table[index])
            hash_table[index]=None
            return
        index = (index+1)%size
    
       
def display(hash_table):
    print(hash_table)
    
def search(key,hash_table,size):
    index=key%size
    
    for _ in range(size):
        
        if key in hash_table[index]:
            print("key found at index : ", index,"keys n values :: ",hash_table[index])
            return
        index=(index+1)%size
        


print("HASH TABLE SIMULATION")
size=int(input("enter the size of hash table :: "))
hash_table=[None for _ in range(size)]


while True:
    print("1. insert value")
    print("2. delete value")
    print("3. display hash table")
    print("4. search key")
    print("5. exit")
    
    c=int(input("enter your choice : "))
    
    if c==1:
        key=int(input("enter the key to insert :: "))

        insert(key,value,hash_table,size)
    
    elif c==2:
        key=int(input("enter the key to delete :: "))
        delete(key,hash_table,size)
        
        
    elif c==3:
        display(hash_table)
        
    elif c==4:
        key=int(input("enter the key to search :: "))
        search(key,hash_table,size)
